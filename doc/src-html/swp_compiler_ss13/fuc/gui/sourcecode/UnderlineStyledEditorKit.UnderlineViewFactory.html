001package swp_compiler_ss13.fuc.gui.sourcecode;
002
003import java.awt.Color;
004import java.awt.Graphics;
005import java.awt.Rectangle;
006import java.awt.Shape;
007
008import javax.swing.text.AbstractDocument;
009import javax.swing.text.AttributeSet;
010import javax.swing.text.Element;
011import javax.swing.text.LabelView;
012import javax.swing.text.Segment;
013import javax.swing.text.StyleConstants;
014import javax.swing.text.StyledEditorKit;
015import javax.swing.text.View;
016import javax.swing.text.ViewFactory;
017
018/**
019 * {@link StyledEditorKit} with function for wavy line ({@link #WAVY_LINE}) and
020 * underline color ({@link #UNDERLINE_COLOR})
021 * 
022 * @author "Eduard Wolf"
023 * 
024 */
025public class UnderlineStyledEditorKit extends StyledEditorKit {
026
027        private static final long serialVersionUID = 1L;
028
029        /**
030         * Key for {@link AttributeSet} to set a {@link Color} for the
031         * {@link StyleConstants#Underline}
032         */
033        public static final Object UNDERLINE_COLOR = "underline-color";
034        /**
035         * Key for {@link AttributeSet} to draw a wavy line with the
036         * {@link #UNDERLINE_COLOR} or if not specified with the default text color
037         * under the Element
038         */
039        public static final Object WAVY_LINE = "wavy line";
040
041        private final ViewFactory defaultFactory;
042
043        public UnderlineStyledEditorKit() {
044                super();
045                defaultFactory = new UnderlineViewFactory(super.getViewFactory());
046        }
047
048        @Override
049        public ViewFactory getViewFactory() {
050                return defaultFactory;
051        }
052
053        static class UnderlineViewFactory implements ViewFactory {
054
055                private final ViewFactory defaultFactury;
056
057                public UnderlineViewFactory(ViewFactory defaultFactury) {
058                        if (defaultFactury == null) {
059                                throw new NullPointerException("defaultFactory cannot be null");
060                        }
061                        this.defaultFactury = defaultFactury;
062                }
063
064                @Override
065                public View create(Element elem) {
066                        if (AbstractDocument.ContentElementName.equals(elem.getName())) {
067                                return new UnderlineColorLabelView(elem);
068                        }
069                        return defaultFactury.create(elem);
070                }
071
072        }
073
074        static class UnderlineColorLabelView extends LabelView {
075
076                public UnderlineColorLabelView(Element elem) {
077                        super(elem);
078                }
079
080                @Override
081                public void paint(Graphics g, Shape a) {
082                        super.paint(g, a);
083                        paintUnderline(g, a);
084                }
085
086                /**
087                 * copy from {@link GlyphView#paintTextUsingColor}
088                 */
089                protected void paintUnderline(Graphics g, Shape a) {
090                        AttributeSet attributes = getElement().getAttributes();
091                        boolean underline = StyleConstants.isUnderline(attributes);
092                        Boolean wavy = (Boolean) attributes.getAttribute(WAVY_LINE);
093                        boolean wavyLine = wavy == null ? false : wavy.booleanValue();
094                        Color newColor = (Color) attributes.getAttribute(UNDERLINE_COLOR);
095                        if (wavyLine || underline && newColor != null) {
096                                if (newColor == null) {
097                                        newColor = StyleConstants.getForeground(attributes);
098                                }
099                                int p0 = getStartOffset();
100                                int p1 = getEndOffset();
101                                // calculate x coordinates
102                                Rectangle alloc = (a instanceof Rectangle) ? (Rectangle) a : a.getBounds();
103                                View parent = getParent();
104                                if ((parent != null) && (parent.getEndOffset() == p1)) {
105                                        // strip whitespace on end
106                                        Segment s = getText(p0, p1);
107                                        while (Character.isWhitespace(s.last())) {
108                                                p1 -= 1;
109                                                s.count -= 1;
110                                        }
111                                }
112                                int x0 = alloc.x;
113                                int p = getStartOffset();
114                                if (p != p0) {
115                                        x0 += (int) getGlyphPainter().getSpan(this, p, p0, getTabExpander(), x0);
116                                }
117                                int x1 = x0 + (int) getGlyphPainter().getSpan(this, p0, p1, getTabExpander(), x0);
118
119                                // calculate y coordinate
120                                int y = alloc.y + alloc.height - (int) getGlyphPainter().getDescent(this) + 1;
121
122                                Color oldColor = g.getColor();
123                                g.setColor(newColor);
124                                if (underline) {
125                                        g.drawLine(x0, y, x1, y);
126                                }
127                                if (wavyLine) {
128                                        int y1 = 0;
129                                        int yChange = 1;
130                                        for (int i = x0; i <= x1; i++, y1 += yChange) {
131                                                g.drawLine(i, y + y1, i, y + y1);
132                                                if (y1 == yChange) {
133                                                        yChange = -yChange;
134                                                }
135                                        }
136                                }
137                                g.setColor(oldColor);
138
139                        }
140                }
141        }
142}
