001package swp_compiler_ss13.fuc.ast;
002
003import java.util.LinkedList;
004import java.util.List;
005
006import org.apache.log4j.Logger;
007
008import swp_compiler_ss13.common.ast.ASTNode;
009import swp_compiler_ss13.common.ast.nodes.ExpressionNode;
010import swp_compiler_ss13.common.ast.nodes.IdentifierNode;
011import swp_compiler_ss13.common.ast.nodes.binary.AssignmentNode;
012
013/**
014 * PrintNode implementation
015 * 
016 * @author "Frank Zechert, Danny Maasch"
017 * @version 1
018 */
019public class AssignmentNodeImpl extends ASTNodeImpl implements AssignmentNode {
020
021        /**
022         * The logger
023         */
024        private static Logger logger = Logger.getLogger(PrintNodeImpl.class);
025
026        /**
027         * left node
028         */
029        private IdentifierNode leftNode;
030
031        /**
032         * right node
033         */
034        private ExpressionNode rightNode;
035
036        @Override
037        public ASTNodeType getNodeType() {
038                return ASTNodeType.AssignmentNode;
039        }
040
041        @Override
042        public Integer getNumberOfNodes() {
043                Integer nodes = 1;
044                if (this.leftNode != null) {
045                        nodes += this.leftNode.getNumberOfNodes();
046                }
047                if (this.rightNode != null) {
048                        nodes += this.rightNode.getNumberOfNodes();
049                }
050                return nodes;
051        }
052
053        @Override
054        public List<ASTNode> getChildren() {
055                List<ASTNode> nodes = new LinkedList<>();
056                if (this.leftNode != null) {
057                        nodes.add(this.leftNode);
058                }
059                if (this.rightNode != null) {
060                        nodes.add(this.rightNode);
061                }
062                return nodes;
063        }
064
065        @Override
066        public IdentifierNode getLeftValue() {
067                if (this.leftNode == null) {
068                        logger.warn("returning null as the left value for the assignment!");
069                }
070                return this.leftNode;
071        }
072
073        @Override
074        public void setLeftValue(IdentifierNode identifier) {
075                if (identifier == null) {
076                        logger.error("The argument identifier can not be null!");
077                        throw new IllegalArgumentException("The argument identifier can not be null!");
078                }
079                
080                identifier.setParentNode(this);
081                
082                this.leftNode = identifier;
083        }
084
085        @Override
086        public ExpressionNode getRightValue() {
087                if (this.rightNode == null) {
088                        logger.warn("returning null as the right value for the assignment!");
089                }
090                return this.rightNode;
091        }
092
093        @Override
094        public void setRightValue(ExpressionNode node) {
095                if (node == null) {
096                        logger.error("The argument identifier can not be null!");
097                        throw new IllegalArgumentException("The argument identifier can not be null!");
098                }
099                
100                node.setParentNode(this);
101                
102                this.rightNode = node;
103        }
104
105}
