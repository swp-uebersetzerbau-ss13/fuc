001package swp_compiler_ss13.common.util;
002
003import java.util.ArrayList;
004import java.util.Collection;
005import java.util.Iterator;
006import java.util.ServiceConfigurationError;
007import java.util.ServiceLoader;
008
009import swp_compiler_ss13.common.backend.Backend;
010import swp_compiler_ss13.common.ir.IntermediateCodeGenerator;
011import swp_compiler_ss13.common.lexer.Lexer;
012import swp_compiler_ss13.common.parser.Parser;
013
014/**
015 * The module provider gives access to the module implementation which is
016 * configured by the ServiceLoader-API.
017 * 
018 * If the configuration of the ServiceLoader-API is not correct, a 
019 * {@link ServiceConfigurationError} may be thrown.
020 * 
021 */
022public class ModuleProvider {
023
024        /**
025         * @return a available Lexer implementation instance
026         * @throws ServiceConfigurationError
027         */
028        public static Lexer getLexerInstance() {
029                return getModuleInstance(Lexer.class);
030        }
031
032        /**
033         * @return a collection of all available Lexer implementation instance
034         * @throws ServiceConfigurationError
035         */
036        public static Collection<? extends Lexer> getAllLexerInstance() {
037                return getAllModuleInstances(Lexer.class);
038        }
039
040        /**
041         * @return a available Parser implementation instance
042         * @throws ServiceConfigurationError
043         */
044        public static Parser getParserInstance() {
045                return getModuleInstance(Parser.class);
046        }
047
048        /**
049         * @return a collection of all available Parser implementation instance
050         * @throws ServiceConfigurationError
051         */
052        public static Collection<? extends Parser> getAllParserInstance() {
053                return getAllModuleInstances(Parser.class);
054        }
055
056        /**
057         * @return a available IntermediateCodeGenerator implementation instance
058         * @throws ServiceConfigurationError
059         */
060        public static IntermediateCodeGenerator getCodeGeneratorInstance() {
061                return getModuleInstance(IntermediateCodeGenerator.class);
062        }
063
064        /**
065         * @return a collection of all available IntermediateCodeGenerator implementation instance
066         * @throws ServiceConfigurationError
067         */
068        public static Collection<? extends IntermediateCodeGenerator> getCodeAllGeneratorInstance() {
069                return getAllModuleInstances(IntermediateCodeGenerator.class);
070        }
071
072        /**
073         * @return a available Backend implementation instance
074         * @throws ServiceConfigurationError
075         */
076        public static Backend getBackendInstance() {
077                return getModuleInstance(Backend.class);
078        }
079
080        /**
081         * @return a collection of all available Backend implementation instance
082         * @throws ServiceConfigurationError
083         */
084        public static Collection<? extends Backend> getAllBackendInstance() {
085                return getAllModuleInstances(Backend.class);
086        }
087
088        /**
089         * simple wrapper for the ServiceLoader of the JDK
090         * @param clazz interface class for which a instance is needed
091         * @return the instance of the first available implementation
092         */
093        public static <T> T getModuleInstance(Class<T> clazz) {
094                Iterator<T> iterator = ServiceLoader.load(clazz).iterator();
095                return iterator.hasNext() ? iterator.next() : null;
096        }
097
098        /**
099         simple wrapper for the ServiceLoader of the JDK
100         * @param clazz interface class for which a instance is needed
101         * @return a collection of instances of each available implementation
102         */
103        public static <T> Collection<? extends T> getAllModuleInstances(
104                        Class<T> clazz) {
105                Collection<T> result = new ArrayList<T>();
106                for (T e : ServiceLoader.load(clazz))
107                        result.add(e);
108                return result;
109        }
110}
